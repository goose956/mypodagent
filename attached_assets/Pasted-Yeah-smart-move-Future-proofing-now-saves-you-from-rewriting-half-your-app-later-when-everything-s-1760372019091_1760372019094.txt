jobs/queue_handler.py

# Simple placeholder queue handler for image generation jobs

queue = []

def add_to_queue(job):
    """Add a job to the queue. Replace later with Redis."""
    queue.append(job)
    print(f"[QUEUE] Added job: {job}")

def process_queue(generate_image_func):
    """Process all jobs in the queue (synchronously for now)."""
    while queue:
        job = queue.pop(0)
        print(f"[WORKER] Processing job: {job}")
        try:
            generate_image_func(job)
        except Exception as e:
            print(f"[ERROR] Failed to process job: {e}")


jobs/worker.py

# Worker script to simulate queue processing

import time
from .queue_handler import process_queue

def dummy_generate_image(job):
    print(f"[KIE API] Generating image for prompt: {job['prompt']}")
    time.sleep(2)
    print(f"[KIE API] Finished image for {job['user_id']}")

if __name__ == "__main__":
    while True:
        process_queue(dummy_generate_image)
        time.sleep(5)  # Check queue every 5 seconds


main.py (example integration)

from jobs.queue_handler import add_to_queue

def handle_image_request(user_id, prompt):
    job = {"user_id": user_id, "prompt": prompt}
    add_to_queue(job)
    return {"status": "queued", "prompt": prompt}

# Example test
if __name__ == "__main__":
    handle_image_request("user123", "a glowing forest with foxes")
